// ========================================== // CRYPTO MARKET CONTROL ENGINE V3 (ALL-IN-ONE) // ========================================== const MarketEngine = (function(){ // ------------------------------ // STATE // ------------------------------ let marketBias = "NEUTRAL"; let lastBias = "NEUTRAL"; let btcDom = 0; let ethDom = 0; let usdtDom = 0; let totalCap = 0; let prevTotalCap = 0; let prevBtcDom = 0; let biasPersistence = 0; // anti-noise counter // ------------------------------ // CORE ENGINE // ------------------------------ async function run(){ try { const res = await fetch("https://api.coingecko.com/api/v3/global"); const json = await res.json(); const market = json.data.market_cap_percentage; // ------------------------------ // DATA UPDATE // ------------------------------ prevTotalCap = totalCap; prevBtcDom = btcDom; btcDom = market.btc; ethDom = market.eth; usdtDom = market.usdt || 0; totalCap = json.data.total_market_cap.usd; // ------------------------------ // DERIVED METRICS // ------------------------------ const ethStrength = btcDom ? (ethDom / btcDom) : 0; const capChange = prevTotalCap ? ((totalCap - prevTotalCap) / prevTotalCap) * 100 : 0; const btcDomChange = prevBtcDom ? btcDom - prevBtcDom : 0; // ------------------------------ // SCORING SYSTEM (SMART FLOW) // ------------------------------ let score = 0; // BTC DOM LEVEL + TREND if (btcDom > 52) score -= 2; else if (btcDom < 48) score += 2; if (btcDomChange > 0.2) score -= 1; // rising dominance = risk off if (btcDomChange < -0.2) score += 1; // falling dominance = alt rotation // USDT DOM (FEAR / SIDELINE MONEY) if (usdtDom > 6.5) score -= 2; else if (usdtDom < 5) score += 1; // ETH VS BTC (ALT STRENGTH) if (ethStrength > 0.08) score += 2; else if (ethStrength < 0.06) score -= 1; // TOTAL MARKET FLOW (LIQUIDITY TREND) if (capChange > 1) score += 2; else if (capChange < -1) score -= 2; // ------------------------------ // RAW BIAS // ------------------------------ let newBias = "NEUTRAL"; if (score >= 3) newBias = "BULLISH"; else if (score <= -2) newBias = "BEARISH"; // ------------------------------ // ANTI-NOISE (PERSISTENCE FILTER) // ------------------------------ if (newBias === lastBias) { biasPersistence++; } else { biasPersistence = 0; } // only confirm after stability if (biasPersistence >= 2) { marketBias = newBias; } lastBias = newBias; // ------------------------------ // UI + DEBUG // ------------------------------ const el = document.getElementById("bias"); if (el) { el.innerText = marketBias; el.style.color = marketBias === "BULLISH" ? "#22c55e" : marketBias === "BEARISH" ? "#ef4444" : "#eab308"; } console.log( "BIAS:", marketBias, "| SCORE:", score, "| BTC.D:", btcDom.toFixed(2), "| ΔBTC.D:", btcDomChange.toFixed(2), "| ETH/BTC:", ethStrength.toFixed(4), "| USDT.D:", usdtDom.toFixed(2), "| ΔTOTAL%:", capChange.toFixed(2), "| PERSIST:", biasPersistence ); } catch (err) { console.log("Engine error:", err); } } // ------------------------------ // START LOOP // ------------------------------ function start(interval = 15000){ run(); setInterval(run, interval); } // ------------------------------ // EXPORT // ------------------------------ return { start }; })(); // ========================================== // RUN ENGINE // ========================================== MarketEngine.start(15000);
