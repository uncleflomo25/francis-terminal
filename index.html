// ===== MAIN CALCULATION FUNCTION ===== function calculateLotSize() { const balance = parseFloat(document.getElementById('accountBalance').value) || 0; const entryPrice = parseFloat(document.getElementById('entryPrice').value) || 0; let slPrice = 0; // ========================= // GET SL PRICE // ========================= if (slMethod === 'manual') { slPrice = parseFloat(document.getElementById('slPriceManual').value) || 0; } if (slMethod === 'points') { slPrice = parseFloat(document.getElementById('slPriceAuto').value) || 0; } if (slMethod === 'lotsize') { slPrice = parseFloat(document.getElementById('slPriceLotAuto').value) || 0; } let tpPrice = parseFloat(document.getElementById('tpPrice').value) || parseFloat(document.getElementById('tpPriceAuto').value) || 0; // ========================= // LOT SIZE // ========================= let lotSize = parseFloat(document.getElementById('slLotSize').value) || parseFloat(document.getElementById('lotSize').value) || 0; if (!balance || !entryPrice || !slPrice || !lotSize) { return; } // ========================= // CALCULATE SL POINTS // ========================= const slPoints = Math.abs(entryPrice - slPrice) / 0.0001; // ========================= // AUTO RISK AED // ========================= const riskAED = slPoints * 0.0001 * lotSize * 100000; // ========================= // AUTO RISK % // ========================= const riskPercentage = (riskAED / balance) * 100; // ========================= // USD CONVERSION // ========================= const riskUSD = riskAED / exchangeRate; // ========================= // DISPLAY // ========================= document.getElementById('displayBalance').textContent = balance.toFixed(2) + ' AED'; document.getElementById('riskAmountAED').textContent = riskAED.toFixed(2) + ' AED'; document.getElementById('riskAmountUSD').textContent = riskUSD.toFixed(2) + ' USD'; document.getElementById('riskMetric').textContent = riskAED.toFixed(0); document.getElementById('slPointsDisplay').textContent = slPoints.toFixed(0) + ' pts'; document.getElementById('slPriceDisplay').textContent = slPrice.toFixed(4); document.getElementById('calculatedLotSize').textContent = lotSize.toFixed(2); // ========================= // AUTO UPDATE RISK % // ========================= document.getElementById('riskPercentage').value = riskPercentage.toFixed(2); // ========================= // POTENTIAL LOSS // ========================= document.getElementById('potentialLossAED').textContent = riskAED.toFixed(2) + ' AED'; document.getElementById('potentialLossUSD').textContent = riskUSD.toFixed(2) + ' USD'; // ========================= // TAKE PROFIT // ========================= let tpPoints = 0; let potentialProfitAED = 0; let potentialProfitUSD = 0; if (tpPrice > 0) { tpPoints = Math.abs(tpPrice - entryPrice) / 0.0001; potentialProfitAED = tpPoints * 0.0001 * lotSize * 100000; potentialProfitUSD = potentialProfitAED / exchangeRate; document.getElementById('tpPointsDisplay').textContent = tpPoints.toFixed(0) + ' pts'; document.getElementById('tpPriceDisplay').textContent = tpPrice.toFixed(4); document.getElementById('potentialProfitAED').textContent = potentialProfitAED.toFixed(2) + ' AED'; document.getElementById('rrRatio').textContent = '1:' + rrRatio.toFixed(2); // ========================= // RATIO BAR // ========================= const fill = Math.min(rrRatio * 25, 100); document.getElementById('ratioFill').style.width = fill + '%'; // ========================= // BREAKEVEN % // ========================= const breakeven = 100 / (1 + rrRatio); document.getElementById('breakevenRate').textContent = breakeven.toFixed(1) + '%'; } else { document.getElementById('tpPointsDisplay').textContent = '0 pts'; document.getElementById('tpPriceDisplay').textContent = '0.0000'; document.getElementById('potentialProfitAED').textContent = '0 AED'; document.getElementById('potentialProfitUSD').textContent = '0 USD'; document.getElementById('rewardMetric').textContent = '0'; document.getElementById('rrRatio').textContent = '1:1'; document.getElementById('breakevenRate').textContent = '50%'; document.getElementById('ratioFill').style.width = '50%'; } }
